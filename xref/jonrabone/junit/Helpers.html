12  /**
13   * Utility functions for working with annotations.
14   */
15  public final class Helpers {
16  
17      private Helpers() {
18      }
19  
20      /**
21       * Check if the suite name matches the list of suites passed in inSuites.
22       *
23       * @param inSuites  annotation to test
24       * @param suiteName suite name to match
25       * @return true if the suite name is found in the values of the inSuites annotation,
26       * false otherwise.
27       */
28      public static boolean matchesSuiteName(final InSuite inSuites, final SuiteName suiteName) {
29          if (inSuites != null) {
30              for (String suite : inSuites.value()) {
31                  if (suiteName.value().equals(suite)) {
32                      return true;
33                  }
34              }
35          }
36          return false;
37      }
38  
39      /**
40       * Check if the class has any method which has an InSuite annotation matching the given suite name.
41       *
42       * @param cls       the class to check.
43       * @param suiteName the suite name to match.
44       * @return true if the suite name is found in the values of the inSuites annotation of any
45       * method on the class, false otherwise.
46       */
47      public static boolean classInSuite(final Class<?> cls, final SuiteName suiteName) {
48          final InSuite classInSuite = cls.getAnnotation(InSuite.class);
49          for (Method method : cls.getMethods()) {
50              if (methodInSuite(method, classInSuite, suiteName)) {
51                  return true;
52              }
53          }
54          return false;
55      }
56  
57      /**
58       * Check if the method has an InSuite annotation matching the given suite name, given any
59       * annotation present on the parent class.
60       *
61       * @param method       the method to check.
62       * @param classInSuite the InSuite annotation on the parent class, or null.
63       * @param suiteName    the suite name to match.
64       * @return true if the suite name is found in the values of the inSuites annotation of the
65       * method and it does not conflict with any annotation on the parent class, false otherwise.
66       */
67      public static boolean methodInSuite(final Method method, final InSuite classInSuite, final SuiteName suiteName) {
68          final boolean classPresent = (classInSuite != null);
69          final boolean classMatches = matchesSuiteName(classInSuite, suiteName);
70          final boolean methodPresent = method.isAnnotationPresent(InSuite.class);
71          final boolean methodMatches = matchesSuiteName(method.getAnnotation(InSuite.class), suiteName);
72          if (classPresent && methodPresent && (classMatches != methodMatches)) {
73              // TODO consider using a logging framework (spit)
74              System.err.println("ignoring mismatched suite annotation on method " + method +
75                      " : entire class marked with suite(s) " + Arrays.toString(classInSuite.value()));
76              return false;
77          }
78          return (classMatches || methodMatches);
79      }
80  
81      /**
82       * Find all classes on the same classpath as a supplied base class with the given annotation. Only class and method
83       * annotations are scanned.
84       *
85       * @param baseClass       class to determine base classpath from.
86       * @param annotationClass annotation to look for.
87       * @return a set of class names.
88       * @throws IOException if an IO error occurred.
89       */
90      public static Set<String> findAnnotatedClasses(final Class<?> baseClass, final Class<?> annotationClass) throws IOException {
91          final AnnotationDB db = new AnnotationDB();
92          db.setScanClassAnnotations(true);
93          db.setScanMethodAnnotations(true);
94          db.setScanFieldAnnotations(false);
95          db.setScanParameterAnnotations(false);
96  
97          db.scanArchives(ClasspathUrlFinder.findClassBase(baseClass));
98  
99          return db.getAnnotationIndex().get(annotationClass.getName());
100     }
101 
102     /**
103      * Filter a set of classnames to only those which are marked with or contain methods marked with
104      * {@link InSuite} annotations matching the given suite name.
105      *
106      * @param suiteName  the suite name to match.
107      * @param classNames class names to scan.
108      * @return a set of filtered classes, may be empty.
109      * @throws ClassNotFoundException if a class cannot be loaded.
110      */
111     public static Set<Class<?>> filterClassesToSuite(final SuiteName suiteName, final Set<String> classNames) throws ClassNotFoundException {
112         final Set<Class<?>> classes = new HashSet<Class<?>>();
113         for (String className : classNames) {
114             final Class<?> cls = Class.forName(className);
115             if (classInSuite(cls, suiteName)) {
116                 classes.add(cls);
117             }
118         }
119         return classes;
120     }
